    z_rev = (row['revenue'] - row['revenue_mean']) / row['revenue_stdev']
    z_tax = (row['tax'] - row['tax_mean']) / row['tax_stdev']
    
    z = sqrt(z_rev**2 + z_tax**2)
    
    return z > 3
